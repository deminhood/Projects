  document.getElementById('s-chars').textContent = totalChars;
  document.getElementById('s-done').textContent = doneChars;
  document.getElementById('h-steps').textContent = doneSteps;
  document.getElementById('h-complete').textContent = doneChars;
  document.getElementById('h-total').textContent = totalChars;
}

function buildPhases() {
  const container = document.getElementById('phases-container');
  PHASES.forEach((phase, pi) => {
    const total = phase.steps.length;
    const done = phase.steps.filter((_, si) => checks[phase.id + '-' + si]).length;
    const pct = Math.round(done / total * 100);
    const isDone = done === total;

    const el = document.createElement('div');
    el.className = 'phase' + (isDone ? ' complete' : '');
    el.id = 'phase-' + phase.id;

    el.innerHTML = `
      <div class="phase-header" onclick="togglePhase('${phase.id}')">
        <div class="phase-badge ${isDone ? 'badge-done' : 'badge-active'}" id="badge-${phase.id}">${isDone ? '✓' : pi + 1}</div>
        <div class="phase-info">
          <div class="phase-title">${phase.title}</div>
          <div class="phase-sub" id="sub-${phase.id}">${done}/${total} steps complete</div>
        </div>
        <div class="phase-arrow" id="arrow-${phase.id}">▶</div>
      </div>
      <div class="phase-body" id="body-${phase.id}">
        <div class="progress-wrap">
          <div class="progress-track">
            <div class="progress-fill" id="bar-${phase.id}" style="width:${pct}%"></div>
          </div>
        </div>
        ${phase.steps.map((step, si) => {
          const key = phase.id + '-' + si;
          const isChecked = !!checks[key];
          return `
          <div class="step">
            <div class="step-checkbox ${isChecked ? 'checked' : ''}" id="chk-${key}" onclick="toggleStep('${key}', '${phase.id}', ${total})">✓</div>
            <div class="step-content">
              <div class="step-title ${isChecked ? 'done' : ''}" id="stitle-${key}">${step.title}</div>
              <div class="step-desc">${step.desc}</div>
              ${step.tip ? `<div class="callout callout-tip"><strong>Tip:</strong> ${step.tip}</div>` : ''}
              ${step.warn ? `<div class="callout callout-warn"><strong>Heads up:</strong> ${step.warn}</div>` : ''}
              ${step.link ? `<a class="step-link" href="${step.link}" target="_blank">${step.linkText}</a>` : ''}
            </div>
          </div>`;
        }).join('')}
      </div>`;
    container.appendChild(el);
  });
}

function togglePhase(id) {
  const body = document.getElementById('body-' + id);
  const arrow = document.getElementById('arrow-' + id);
  const open = body.classList.contains('open');
  body.classList.toggle('open', !open);
  arrow.classList.toggle('open', !open);
}

function toggleTracker() {
  const body = document.getElementById('tracker-body');
  const arrow = document.getElementById('tracker-arrow');
  body.classList.toggle('open');
  arrow.classList.toggle('open');
}

function toggleStep(key, phaseId, total) {
  checks[key] = !checks[key];
  persist();

  const chk = document.getElementById('chk-' + key);
  const title = document.getElementById('stitle-' + key);
  chk.classList.toggle('checked', checks[key]);
  title.classList.toggle('done', checks[key]);

  const done = PHASES.find(p => p.id === phaseId).steps.filter((_, si) => checks[phaseId + '-' + si]).length;
  const isDone = done === total;
  const pct = Math.round(done / total * 100);

  document.getElementById('sub-' + phaseId).textContent = done + '/' + total + ' steps complete';
  document.getElementById('bar-' + phaseId).style.width = pct + '%';

  const badge = document.getElementById('badge-' + phaseId);
  const pi = PHASES.findIndex(p => p.id === phaseId);
  badge.className = 'phase-badge ' + (isDone ? 'badge-done' : 'badge-active');
  badge.textContent = isDone ? '✓' : (pi + 1);

  const phaseEl = document.getElementById('phase-' + phaseId);
  phaseEl.classList.toggle('complete', isDone);

  updateStats();
}

function renderChars() {
  const grid = document.getElementById('char-grid');
  if (chars.length === 0) {
    grid.innerHTML = '<div class="empty-state">No characters yet — hit "Add character" to map your first figure to a story.</div>';
    return;
  }
  const tagLabel = { pending: 'tag pending', writing: 'writing tag', done: 'tag done ✓' };
  const tagClass = { pending: 'pill-gray', writing: 'pill-blue', done: 'pill-green' };
  const typeLabel = { '3d': '3D print', clay: 'clay', existing: 'existing toy', other: 'other' };

  grid.innerHTML = chars.map((c, i) => `
    <div class="char-card ${c.tag === 'done' ? 'complete' : ''}">
      <button class="del-btn" onclick="deleteChar(${i})" title="Remove">×</button>
      <div class="char-name">${c.name}</div>
      <div class="char-story">${c.story || 'No file mapped yet'}</div>
      <div class="char-pills">
        <span class="pill ${tagClass[c.tag]}">${tagLabel[c.tag]}</span>
        <span class="pill pill-gray">${typeLabel[c.type] || c.type}</span>
      </div>
    </div>`).join('');

  updateStats();
}

function openModal() {
  document.getElementById('modal-overlay').classList.add('open');
  setTimeout(() => document.getElementById('m-name').focus(), 50);
}

function closeModal() {
  document.getElementById('modal-overlay').classList.remove('open');
  document.getElementById('m-name').value = '';
  document.getElementById('m-story').value = '';
  document.getElementById('m-type').value = '3d';
  document.getElementById('m-tag').value = 'pending';
}

function saveChar() {
  const name = document.getElementById('m-name').value.trim();
  if (!name) { document.getElementById('m-name').focus(); return; }
  chars.push({
    name,
    story: document.getElementById('m-story').value.trim(),
    type: document.getElementById('m-type').value,
    tag: document.getElementById('m-tag').value
  });
  persist();
  renderChars();
  closeModal();
}

function deleteChar(i) {
  if (!confirm('Remove ' + chars[i].name + '?')) return;
  chars.splice(i, 1);
  persist();
  renderChars();
}

document.getElementById('modal-overlay').addEventListener('click', function(e) {
  if (e.target === this) closeModal();
});

document.addEventListener('keydown', e => {
  if (e.key === 'Escape') closeModal();
});

buildPhases();
renderChars();
updateStats();
